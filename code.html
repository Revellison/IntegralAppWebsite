Примеры кода

Базовый пример

const integral = require('integral-platform');

const app = integral.createApp({
    name: 'MyApp',
    version: '1.0.0'
});

app.start();

Расширенная конфигурация

const config = {
    name: 'MyApp',
    version: '1.0.0',
    plugins: [
        'integral-auth',
        'integral-database'
    ],
    settings: {
        port: 3000,
        debug: true
    }
};

const app = integral.createApp(config);
